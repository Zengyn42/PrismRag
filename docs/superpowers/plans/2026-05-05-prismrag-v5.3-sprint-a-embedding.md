```python
def _detect_model_device(model: str, host: str) -> str:
    """Query Ollama /api/ps to determine if model is on GPU or CPU.

    Returns "gpu" if size_vram > 0, "cpu" if loaded but no VRAM, "unknown" otherwise.
    """
    import urllib.request as _req
    url = f"{host.rstrip('/')}/api/ps"
    try:
        with _req.urlopen(url, timeout=5) as resp:
            data = json.loads(resp.read())
        for entry in data.get("models", []):
            if model in entry.get("name", ""):
                return "gpu" if entry.get("size_vram", 0) > 0 else "cpu"
    except Exception:
        pass
    return "unknown"
```

```python
def _content_sha(content: str) -> str:
    """Return the SHA-256 hex digest of the node content string."""
    return hashlib.sha256(content.encode()).hexdigest()


def _load_embed_cache(cache_path: Path) -> dict[str, dict]:
    """Load embed_cache.jsonl into a dict keyed by node_id.

    Each value is {"sha": str, "vec": list[float]}.
    Missing or empty file returns {}.
    Duplicate node_ids: last entry wins (append-only log semantics).
    """
    cache: dict[str, dict] = {}
    if not cache_path.exists():
        return cache
    for line in cache_path.read_text().splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            entry = json.loads(line)
            cache[entry["node_id"]] = {"sha": entry["sha"], "vec": entry["vec"]}
        except (json.JSONDecodeError, KeyError):
            pass
    return cache


def _append_cache_entry(
    cache_path: Path,
    node_id: str,
    sha: str,
    vec: list[float],
) -> None:
    """Append one entry to embed_cache.jsonl atomically.

    Uses 'a' mode — safe for concurrent appends on Linux (each write < PIPE_BUF
    is atomic at OS level for local filesystems).
    """
    line = json.dumps({"node_id": node_id, "sha": sha, "vec": vec}) + "\n"
    with open(cache_path, "a") as fh:
        fh.write(line)
```

```python
def _compute_embeddings_ollama(
    graph: KnowledgeGraph,
    settings: PrismRagSettings | None = None,
    *,
    cache_path: Path | None = None,
) -> dict[str, list[float]]:
    """Compute embeddings using local Ollama (batched), with resume cache support.

    If cache_path is provided, nodes whose content SHA matches the cache are
    skipped (returned from cache). New embeddings are appended to the cache
    immediately after each successful batch so progress survives interrupts.
    """
    nodes_to_embed = _get_embeddable_nodes(graph)
    if not nodes_to_embed:
        logger.info("[embedder/ollama] no embeddable nodes found")
        return {}

    model = settings.ollama_model if settings else _OLLAMA_DEFAULT_MODEL
    host = settings.ollama_host if settings else _OLLAMA_DEFAULT_HOST
    embedder = OllamaEmbedder(model=model, base_url=host, timeout=_OLLAMA_TIMEOUT)
    total = len(nodes_to_embed)

    # ── GPU detection ────────────────────────────────────────────────────────
    device = _detect_model_device(embedder.model, host)
    if device == "cpu":
        logger.warning(
            f"[embedder/ollama] model '{embedder.model}' is running on CPU (VRAM=0). "
            f"Embedding {total} nodes will be very slow. "
            f"Wait for GPU to be free, then restart Ollama to reload on GPU."
        )
    elif device == "gpu":
        logger.info(f"[embedder/ollama] model '{embedder.model}' confirmed on GPU.")

    # ── Load cache ───────────────────────────────────────────────────────────
    cache = _load_embed_cache(cache_path) if cache_path else {}
    vectors: dict[str, list[float]] = {}
    to_embed: list[tuple[str, str]] = []   # (node_id, content) pairs that need embedding

    for node_id, content in nodes_to_embed:
        sha = _content_sha(content)
        if node_id in cache and cache[node_id]["sha"] == sha:
            vectors[node_id] = cache[node_id]["vec"]
        else:
            to_embed.append((node_id, content))

    cached_count = len(vectors)
    if cached_count:
        logger.info(f"[embedder/ollama] cache hit: {cached_count}/{total} nodes skipped")

    remaining = len(to_embed)
    logger.info(
        f"[embedder/ollama] computing {remaining} embeddings "
        f"(model={embedder.model}, batch={_OLLAMA_BATCH_SIZE})"
    )

    # ── Embed remaining nodes ────────────────────────────────────────────────
    for batch_start in range(0, remaining, _OLLAMA_BATCH_SIZE):
        batch = to_embed[batch_start: batch_start + _OLLAMA_BATCH_SIZE]
        node_ids = [nid for nid, _ in batch]
        texts = [_truncate(content) for _, content in batch]
        try:
            vecs = embedder.embed_batch(texts)
            for nid, content, vec in zip(node_ids, [c for _, c in batch], vecs):
                vectors[nid] = vec
                if cache_path is not None:
                    _append_cache_entry(cache_path, nid, _content_sha(content), vec)
        except Exception as exc:
            logger.error(
                f"[embedder/ollama] batch {batch_start}–{batch_start + len(batch) - 1} failed: {exc}"
                f" — retrying one-by-one"
            )
            for nid, content in batch:
                try:
                    vec = embedder.embed_query(_truncate(content))
                    vectors[nid] = vec
                    if cache_path is not None:
                        _append_cache_entry(cache_path, nid, _content_sha(content), vec)
                except Exception as exc2:
                    logger.error(f"[embedder/ollama] node {nid} failed: {exc2}")

        done = min(batch_start + _OLLAMA_BATCH_SIZE, remaining)
        if done % 160 == 0 or done == remaining:
            logger.info(f"[embedder/ollama] progress: {cached_count + done}/{total}")

    logger.info(f"[embedder/ollama] done: {len(vectors)}/{total}")
    return vectors
```
