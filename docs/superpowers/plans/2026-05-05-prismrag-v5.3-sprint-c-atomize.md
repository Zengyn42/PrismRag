```python
"""atomize_document tool suite — three-phase protocol for document atomization.

Phase 1: atomize_scan   — read document structure (no writes)
Phase 2: atomize_propose — assign KNOW-IDs, write proposal file
Phase 3: atomize_apply   — commit: create knowledge nodes, patch original, update graph

The LLM (Jei) provides all semantic judgment between phases 1 and 2.
PrismRag only does mechanical operations.
"""
from __future__ import annotations

import hashlib
import json
import re
import unicodedata
from datetime import datetime, timezone
from pathlib import Path
from typing import Any

import frontmatter as fm


def _parse_sections(content: str) -> list[dict[str, Any]]:
    """Extract markdown heading sections with line numbers and metadata."""
    lines = content.splitlines()
    sections = []
    i = 0
    while i < len(lines):
        m = re.match(r'^(#{1,6})\s+(.+)$', lines[i])
        if m:
            level = len(m.group(1))
            heading = m.group(2).strip()
            line_start = i + 1  # 1-indexed
            # Find end: next heading of same or higher level, or EOF
            j = i + 1
            while j < len(lines):
                nm = re.match(r'^(#{1,6})\s+', lines[j])
                if nm and len(nm.group(1)) <= level:
                    break
                j += 1
            line_end = j   # exclusive, 1-indexed end is j (next heading's line)
            body = "\n".join(lines[i + 1:j]).strip()
            sections.append({
                "heading": heading,
                "level": level,
                "line_start": line_start,
                "line_end": line_end,
                "char_count": len(body),
                "has_knowledge_refs": bool(re.search(r'\[\[KNOW-\d{4}', body)),
                "first_200_chars": body[:200],
            })
        i += 1
    return sections


def _slug(title: str) -> str:
    """Convert a title to a lowercase hyphenated slug for filenames."""
    s = unicodedata.normalize("NFKD", title).encode("ascii", "ignore").decode()
    s = re.sub(r"[^\w\s-]", "", s).strip().lower()
    s = re.sub(r"[\s_]+", "-", s)
    return s[:40]


def atomize_scan_impl(file_path: Path, vault_root: Path) -> dict[str, Any]:
    """Read document structure and return sections + existing KNOW-IDs.

    No writes. Safe to call multiple times.
    """
    fp = file_path.expanduser().resolve()
    vr = vault_root.expanduser().resolve()
    rel = fp.relative_to(vr)

    post = fm.load(str(fp))
    content = post.content
    frontmatter_data = dict(post.metadata)

    already_atomized: list[str] = frontmatter_data.get("atomized_nodes", [])
    if isinstance(already_atomized, str):
        already_atomized = [already_atomized]

    sections = _parse_sections(content)

    return {
        "path": rel.as_posix(),
        "node_id": rel.with_suffix("").as_posix(),
        "already_atomized": already_atomized,
        "sections": sections,
    }
```

```python
def atomize_propose_impl(
    source_path: str,
    namespace: str,
    claims: list[dict[str, Any]],
    *,
    data_dir: Path,
) -> dict[str, Any]:
    """Assign KNOW-IDs to claims and write a proposal file.

    Called after the LLM has decided what to atomize.
    Returns proposal_id and the list of (title → KNOW-ID) assignments.
    Does NOT modify the vault.
    """
    from prism_rag.store.registry import Registry

    registry = Registry(data_dir / "registry.json")
    pending_dir = data_dir / "atomize-proposals" / "pending"
    pending_dir.mkdir(parents=True, exist_ok=True)

    # Generate stable proposal ID from source + timestamp
    ts = datetime.now(timezone.utc).strftime("%Y-%m-%dT%H-%M-%S")
    src_slug = _slug(source_path.replace("/", "-").replace(".md", ""))
    proposal_id = f"atomize-{ts}-{src_slug}"
    proposal_path = pending_dir / f"{proposal_id}.json"

    # Assign KNOW-IDs to each claim
    enriched_claims = []
    assignments = []
    for claim in claims:
        kid = registry.alloc_id()
        enriched = dict(claim)
        enriched["knowledge_id"] = kid
        enriched_claims.append(enriched)
        assignments.append({"title": claim["title"], "knowledge_id": kid})

    proposal = {
        "id": proposal_id,
        "source_path": source_path,
        "namespace": namespace,
        "created_at": datetime.now(timezone.utc).isoformat(),
        "status": "pending",
        "claims": enriched_claims,
    }

    proposal_path.write_text(json.dumps(proposal, indent=2, ensure_ascii=False))

    return {
        "proposal_id": proposal_id,
        "proposal_path": str(proposal_path),
        "assignments": assignments,
        "total_claims": len(claims),
    }
```

```python
def atomize_apply_impl(
    proposal_id: str,
    *,
    vault_root: Path,
    data_dir: Path,
    skip_graph_update: bool = False,
) -> dict[str, Any]:
    """Apply an approved proposal: create knowledge nodes, patch source doc, update graph.

    Safe to re-run after a crash — skips knowledge files that already exist (idempotent).
    Moves proposal to applied/ only if all steps succeed.
    """
    pending_dir = data_dir / "atomize-proposals" / "pending"
    applied_dir = data_dir / "atomize-proposals" / "applied"
    applied_dir.mkdir(parents=True, exist_ok=True)

    proposal_path = pending_dir / f"{proposal_id}.json"
    if not proposal_path.exists():
        # Check if already applied
        applied_path = applied_dir / f"{proposal_id}.json"
        if applied_path.exists():
            return {"error": f"Proposal '{proposal_id}' already applied", "status": "already_applied"}
        raise FileNotFoundError(f"Proposal not found: {proposal_path}")

    proposal = json.loads(proposal_path.read_text())
    source_path = vault_root / proposal["source_path"]
    knowledge_dir = vault_root / "knowledge"
    knowledge_dir.mkdir(parents=True, exist_ok=True)

    created_nodes: list[str] = []
    patch_instructions: list[dict] = []  # {source_lines, replacement}

    # Phase 1: create knowledge node files
    for claim in proposal["claims"]:
        kid = claim["knowledge_id"]
        filename = f"{kid}-{_slug(claim['title'])}.md"
        node_path = knowledge_dir / filename

        # Idempotent: skip if already written
        if not node_path.exists():
            fm_data = {
                "knowledge_id": kid,
                "title": claim["title"],
                "type": claim.get("type", "fact"),
                "status": "active",
                "created": datetime.now(timezone.utc).date().isoformat(),
            }
            if claim.get("relations"):
                fm_data["relations"] = claim["relations"]

            node_content = f"---\n"
            for k, v in fm_data.items():
                if isinstance(v, list):
                    node_content += f"{k}:\n" + "".join(f"  - {i}\n" for i in v)
                else:
                    node_content += f"{k}: {v}\n"
            node_content += f"---\n\n{claim['body']}\n"
            node_path.write_text(node_content)

        created_nodes.append(kid)
        patch_instructions.append({
            "source_lines": claim["source_lines"],
            "replacement": f"\n> 详见 [[{kid}]]\n",
            "kid": kid,
        })

    # Phase 2: patch original document (apply patches bottom-up to preserve line numbers)
    if source_path.exists():
        lines = source_path.read_text().splitlines(keepends=True)
        # Sort patches by start line descending (bottom-up)
        for patch in sorted(patch_instructions, key=lambda p: p["source_lines"][0], reverse=True):
            start, end = patch["source_lines"]
            # Convert 1-indexed inclusive range to 0-indexed exclusive
            s = max(0, start - 1)
            e = min(len(lines), end)
            lines[s:e] = [patch["replacement"]]
        source_path.write_text("".join(lines))

    # Phase 3: update source doc frontmatter with atomized_nodes
    try:
        post = fm.load(str(source_path))
        existing = post.metadata.get("atomized_nodes", [])
        if isinstance(existing, str):
            existing = [existing]
        post.metadata["atomized_nodes"] = sorted(set(existing) | set(created_nodes))
        source_path.write_text(fm.dumps(post))
    except Exception:
        pass  # Frontmatter update failure is non-fatal

    # Phase 4: incremental graph update
    if not skip_graph_update:
        try:
            from prism_rag.ingest.incremental import ingest_file
            from prism_rag.config import PrismRagSettings
            settings = PrismRagSettings()
            settings.vault_path = vault_root
            settings.data_dir = data_dir
            for kid in created_nodes:
                kid_files = list(knowledge_dir.glob(f"{kid}-*.md"))
                if kid_files:
                    ingest_file(kid_files[0], settings=settings, skip_embed=True, skip_leiden=True, skip_persist=True)
            ingest_file(source_path, settings=settings, skip_embed=True, skip_leiden=True)
        except Exception as exc:
            pass  # Graph update failure is non-fatal; note in result

    # Phase 5: move proposal to applied
    applied_path = applied_dir / f"{proposal_id}.json"
    proposal["status"] = "applied"
    proposal["applied_at"] = datetime.now(timezone.utc).isoformat()
    applied_path.write_text(json.dumps(proposal, indent=2, ensure_ascii=False))
    proposal_path.unlink()

    return {
        "created_nodes": created_nodes,
        "patched_document": proposal["source_path"],
        "proposal_id": proposal_id,
    }
```

```python
@mcp.tool()
def atomize_scan(path: str, namespace: str = "") -> str:
    """Read a vault document's structure to prepare for atomization.

    Returns sections (heading, level, line range, char count, first 200 chars),
    already-atomized KNOW-IDs from frontmatter, and related graph nodes.

    Call this first. Use the output to decide which sections to atomize,
    then call atomize_propose with your decisions.
    """
    settings = _get_settings()
    # Resolve vault_root from namespace or primary
    sources = settings.resolved_graphs
    if namespace:
        src = next((s for s in sources if s.namespace == namespace), None)
        vault_root = src.vault_path.expanduser().resolve() if src else settings.vault_path.expanduser().resolve()
    else:
        vault_root = settings.vault_path.expanduser().resolve()

    fp = (vault_root / path).expanduser().resolve()
    try:
        result = atomize_scan_impl(fp, vault_root=vault_root)
        return json.dumps(result, indent=2, ensure_ascii=False)
    except FileNotFoundError:
        return json.dumps({"error": f"File not found: {path}"})


@mcp.tool()
def atomize_propose(
    path: str,
    namespace: str,
    claims: list[dict],
) -> str:
    """Submit LLM-decided atomization claims and receive KNOW-ID assignments.

    Each claim in `claims` must have:
      - title: str (< 30 chars, declarative)
      - type: str  (fact|concept|decision|rule|procedure|relation)
      - source_section: str  (heading name from atomize_scan)
      - source_lines: [start, end]  (1-indexed, from atomize_scan)
      - body: str  (the extracted atomic content)
      - relations: list[{type, target_id}]  (optional)
      - confidence: str  (high|medium — only submit high/medium)

    Returns proposal_id and list of {title → knowledge_id} assignments.
    Review the assignments, then call atomize_apply(proposal_id) to commit.
    """
    settings = _get_settings()
    sources = settings.resolved_graphs
    if namespace:
        src = next((s for s in sources if s.namespace == namespace), None)
        data_dir = src.data_dir.expanduser().resolve() if src else settings.data_dir.expanduser().resolve()
    else:
        data_dir = settings.data_dir.expanduser().resolve()

    try:
        result = atomize_propose_impl(path, namespace, claims, data_dir=data_dir)
        return json.dumps(result, indent=2, ensure_ascii=False)
    except Exception as exc:
        return json.dumps({"error": str(exc)})


@mcp.tool()
def atomize_apply(proposal_id: str, namespace: str = "") -> str:
    """Commit an approved proposal: create knowledge node files, patch original document.

    Idempotent — safe to call again after a crash. Knowledge files that already exist
    are kept as-is; the original document is patched only for un-patched claims.

    After this call, the proposal moves from pending/ to applied/.
    """
    settings = _get_settings()
    sources = settings.resolved_graphs
    if namespace:
        src = next((s for s in sources if s.namespace == namespace), None)
        vault_root = src.vault_path.expanduser().resolve() if src else settings.vault_path.expanduser().resolve()
        data_dir = src.data_dir.expanduser().resolve() if src else settings.data_dir.expanduser().resolve()
    else:
        vault_root = settings.vault_path.expanduser().resolve()
        data_dir = settings.data_dir.expanduser().resolve()

    try:
        result = atomize_apply_impl(proposal_id, vault_root=vault_root, data_dir=data_dir)
        return json.dumps(result, indent=2, ensure_ascii=False)
    except FileNotFoundError as exc:
        return json.dumps({"error": str(exc)})
```

```python
"""CLI group for managing atomize proposals offline."""
from __future__ import annotations

import json
from pathlib import Path

import typer

atomize_app = typer.Typer(name="atomize", help="Manage atomize_document proposals.")


def _pending_dir(data_dir: Path) -> Path:
    return data_dir / "atomize-proposals" / "pending"


def _applied_dir(data_dir: Path) -> Path:
    return data_dir / "atomize-proposals" / "applied"


@atomize_app.command("list")
def atomize_list(
    data_dir: Path = typer.Option(None, "--data-dir", "-d"),
) -> None:
    """List all pending atomize proposals."""
    from prism_rag.config import PrismRagSettings
    settings = PrismRagSettings()
    dd = (data_dir or settings.data_dir).expanduser().resolve()
    pending = _pending_dir(dd)
    if not pending.exists():
        typer.echo("No pending proposals.")
        return
    proposals = list(pending.glob("*.json"))
    if not proposals:
        typer.echo("No pending proposals.")
        return
    for p in sorted(proposals):
        data = json.loads(p.read_text())
        claim_count = len(data.get("claims", []))
        typer.echo(f"  {data['id']}  source={data['source_path']}  claims={claim_count}")


@atomize_app.command("show")
def atomize_show(
    proposal_id: str = typer.Argument(...),
    data_dir: Path = typer.Option(None, "--data-dir", "-d"),
) -> None:
    """Show details of a pending proposal."""
    from prism_rag.config import PrismRagSettings
    settings = PrismRagSettings()
    dd = (data_dir or settings.data_dir).expanduser().resolve()
    proposal_path = _pending_dir(dd) / f"{proposal_id}.json"
    if not proposal_path.exists():
        typer.echo(f"Proposal not found: {proposal_id}", err=True)
        raise typer.Exit(1)
    data = json.loads(proposal_path.read_text())
    typer.echo(json.dumps(data, indent=2, ensure_ascii=False))


@atomize_app.command("apply")
def atomize_apply_cmd(
    proposal_id: str = typer.Argument(...),
    data_dir: Path = typer.Option(None, "--data-dir", "-d"),
    vault: Path = typer.Option(None, "--vault", "-v"),
) -> None:
    """Apply a pending proposal to the vault."""
    from prism_rag.config import PrismRagSettings
    from prism_rag.ingest.atomize import atomize_apply_impl
    settings = PrismRagSettings()
    dd = (data_dir or settings.data_dir).expanduser().resolve()
    vr = (vault or settings.vault_path).expanduser().resolve()
    result = atomize_apply_impl(proposal_id, vault_root=vr, data_dir=dd)
    typer.echo(json.dumps(result, indent=2, ensure_ascii=False))
```
