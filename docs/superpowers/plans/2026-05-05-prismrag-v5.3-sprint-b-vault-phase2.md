```python
"""KNOW-ID allocator for Vault Phase 2 atomic knowledge nodes.

Allocates monotonically increasing IDs in KNOW-NNNN format.
Thread-safe via fcntl file lock (Linux / macOS).
IDs are never reused — once allocated, a KNOW-ID is permanent even if the node is deleted.
"""
from __future__ import annotations

import fcntl
import json
from datetime import datetime, timezone
from pathlib import Path


class Registry:
    """Persistent KNOW-ID allocator backed by a JSON file.

    Usage::

        reg = Registry(data_dir / "registry.json")
        kid = reg.alloc_id()   # "KNOW-0001"
    """

    SCHEMA = "v1"

    def __init__(self, path: Path) -> None:
        self._path = path

    def alloc_id(self) -> str:
        """Allocate the next KNOW-ID atomically.

        Acquires an exclusive file lock, reads current state, increments,
        writes back, releases lock. Safe for concurrent callers in separate
        processes.
        """
        self._path.parent.mkdir(parents=True, exist_ok=True)
        lock_path = self._path.with_suffix(".lock")
        with open(lock_path, "w") as lock_fh:
            fcntl.flock(lock_fh, fcntl.LOCK_EX)
            try:
                data = self._read()
                next_id = data["next_id"]
                kid = f"KNOW-{next_id:04d}"
                data["next_id"] = next_id + 1
                data["last_assigned"] = kid
                data["updated_at"] = datetime.now(timezone.utc).isoformat()
                self._write(data)
            finally:
                fcntl.flock(lock_fh, fcntl.LOCK_UN)
        return kid

    def peek_next(self) -> str:
        """Return what the next alloc_id() would return, without allocating."""
        data = self._read()
        return f"KNOW-{data['next_id']:04d}"

    def _read(self) -> dict:
        if not self._path.exists():
            return {"schema": self.SCHEMA, "next_id": 1, "last_assigned": None, "updated_at": None}
        return json.loads(self._path.read_text())

    def _write(self, data: dict) -> None:
        from prism_rag.utils.io import atomic_write
        atomic_write(self._path, json.dumps(data, indent=2))
```

```python
@mcp.tool()
def alloc_knowledge_id(namespace: str = "") -> str:
    """Allocate the next KNOW-ID from the registry for this namespace.

    Creates the registry at data_dir/registry.json if it doesn't exist.
    IDs are monotonically increasing and never reused.

    Returns: {"knowledge_id": "KNOW-0001", "registry_path": "..."}
    """
    settings = _get_settings()
    # Resolve data_dir: if namespace given, use its source; else use primary data_dir.
    sources = settings.resolved_graphs
    if namespace:
        src = next((s for s in sources if s.namespace == namespace), None)
        if src is None:
            return json.dumps({"error": f"Namespace '{namespace}' not found"})
        data_dir = src.data_dir.expanduser().resolve()
    else:
        data_dir = settings.data_dir.expanduser().resolve()

    registry = Registry(data_dir / "registry.json")
    kid = registry.alloc_id()
    return json.dumps({"knowledge_id": kid, "registry_path": str(registry._path)})
```

```python
@mcp.tool()
def list_knowledge_nodes(
    namespace: str = "",
    ktype: str = "",
    status: str = "active",
) -> str:
    """List all KNOW-* knowledge nodes in the graph.

    Args:
        namespace: Filter by namespace (default: primary namespace).
        ktype: Filter by ontology type (fact/decision/concept/rule/procedure/relation).
               Empty string = all types.
        status: Filter by status field in frontmatter (default: "active").
                Use "" to return all statuses.

    Returns: {"total": N, "nodes": [{"knowledge_id", "label", "ontology_type", "status"}, ...]}
    """
    fg = _ensure_federated()

    # Collect nodes from relevant namespaces
    namespaces = [namespace] if namespace else fg.namespaces
    results = []

    for ns in namespaces:
        graph = fg.get_graph(ns)
        if graph is None:
            continue
        for node_id, attrs in graph.g.nodes(data=True):
            if attrs.get("kind") != "knowledge":
                continue
            kid = attrs.get("knowledge_id") or attrs.get("metadata", {}).get("know_id")
            if not kid:
                continue
            node_status = attrs.get("frontmatter", {}).get("status", "active")
            if status and node_status != status:
                continue
            node_type = attrs.get("ontology_type") or attrs.get("frontmatter", {}).get("type", "")
            if ktype and node_type != ktype:
                continue
            results.append({
                "knowledge_id": kid,
                "node_id": node_id,
                "label": attrs.get("label", ""),
                "ontology_type": node_type,
                "status": node_status,
                "namespace": ns,
            })

    return json.dumps({"total": len(results), "nodes": results}, indent=2)
```
